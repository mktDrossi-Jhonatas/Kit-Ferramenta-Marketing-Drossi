import os
import subprocess
import tkinter as tk
from tkinter import filedialog, messagebox, ttk

FFMPEG = "ffmpeg"
FFPROBE = "ffprobe"

class Compressor:

    def __init__(self, root):

        self.root = root
        self.root.title("Video Compressor")
        self.root.geometry("650x500")

        self.files = []

        ttk.Label(root, text="Tamanho Final (MB)").pack(pady=10)

        self.size_entry = ttk.Entry(root)
        self.size_entry.insert(0, "30")
        self.size_entry.pack()

        ttk.Button(root, text="Selecionar Vídeos",
                   command=self.select_files).pack(pady=10)

        self.listbox = tk.Listbox(root, width=80, height=15)
        self.listbox.pack()

        ttk.Button(root,
                   text="Comprimir",
                   command=self.compress_all).pack(pady=15)

        self.progress = ttk.Progressbar(root,
                                        orient="horizontal",
                                        mode="determinate",
                                        length=500)

        self.progress.pack()

    def select_files(self):

        files = filedialog.askopenfilenames(
            filetypes=[("Videos",
                        "*.mp4 *.mov *.avi *.mkv *.wmv *.flv")]
        )

        if files:

            self.files = files

            self.listbox.delete(0, tk.END)

            for f in files:
                self.listbox.insert(tk.END, os.path.basename(f))

    def get_duration(self, file):

        cmd = [
            FFPROBE,
            "-v","error",
            "-show_entries","format=duration",
            "-of","default=noprint_wrappers=1:nokey=1",
            file
        ]

        result = subprocess.run(cmd,
                                capture_output=True,
                                text=True)

        return float(result.stdout.strip())

    def compress(self, file, target_mb):

        duration = self.get_duration(file)

        target_size_bits = target_mb * 1024 * 1024 * 8

        audio_bitrate = 128000

        video_bitrate = int(
            (target_size_bits / duration) - audio_bitrate
        )

        if video_bitrate < 100000:
            video_bitrate = 100000

        output_folder = os.path.join(
            os.path.dirname(file),
            "Comprimidos"
        )

        os.makedirs(output_folder, exist_ok=True)

        output = os.path.join(
            output_folder,
            os.path.splitext(os.path.basename(file))[0]
            + "_comp.mp4"
        )

        cmd = [
            FFMPEG,
            "-y",
            "-i", file,
            "-c:v", "libx264",
            "-b:v", str(video_bitrate),
            "-c:a", "aac",
            "-b:a", "128k",
            output
        ]

        subprocess.run(cmd)

    def compress_all(self):

        if not self.files:
            messagebox.showwarning("Aviso","Selecione vídeos.")
            return

        target = float(self.size_entry.get())

        total = len(self.files)

        self.progress["maximum"] = total

        for i,file in enumerate(self.files):

            self.compress(file,target)

            self.progress["value"] = i+1

            self.root.update()

        messagebox.showinfo(
            "Concluído",
            "Todos os vídeos foram comprimidos."
        )


root = tk.Tk()

Compressor(root)

root.mainloop()
